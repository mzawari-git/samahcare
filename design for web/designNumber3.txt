    <script>
        // إعداد الهيدر عند التمرير
        const header = document.getElementById('header');
        const marquee = document.querySelector('.marquee-container');
        
        window.addEventListener('scroll', () => {
            if (window.scrollY > 40) {
                header.classList.add('bg-pure-cream/90', 'backdrop-blur-md', 'shadow-sm', 'py-4');
                header.classList.remove('py-6');
                if(marquee) marquee.style.display = 'none'; // إخفاء الماركي عند النزول
                header.style.top = '0';
            } else {
                header.classList.remove('bg-pure-cream/90', 'backdrop-blur-md', 'shadow-sm', 'py-4');
                header.classList.add('py-6');
                if(marquee && window.innerWidth >= 768) marquee.style.display = 'block';
                header.style.top = 'auto';
            }
        });

        // القائمة الجانبية للموبايل
        const menuBtn = document.getElementById('menu-btn');
        const fullMenu = document.getElementById('full-menu');
        const menuLinks = document.querySelectorAll('.menu-link');
        const lines = menuBtn.querySelectorAll('span');
        let menuOpen = false;

        menuBtn.addEventListener('click', () => {
            menuOpen = !menuOpen;
            if (menuOpen) {
                fullMenu.classList.add('open');
                // تغيير شكل الزر إلى (X)
                lines[0].style.transform = 'rotate(45deg) translate(5px, 5px)';
                lines[1].style.transform = 'rotate(-45deg) translate(2px, -2px)';
                lines[1].style.width = '1.5rem';
                document.body.style.overflow = 'hidden';
            } else {
                closeMenu();
            }
        });

        function closeMenu() {
            menuOpen = false;
            fullMenu.classList.remove('open');
            lines[0].style.transform = 'none';
            lines[1].style.transform = 'none';
            document.body.style.overflow = 'auto';
        }

        menuLinks.forEach(link => link.addEventListener('click', closeMenu));

        // مراقب التمرير (Intersection Observer) لظهور العناصر والصور
        const observerOptions = {
            root: null,
            rootMargin: '0px',
            threshold: 0.2
        };

        const observer = new IntersectionObserver((entries, observer) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('active');
                    observer.unobserve(entry.target);
                }
            });
        }, observerOptions);

        // مراقبة النصوص
        document.querySelectorAll('.reveal').forEach((elem) => {
            observer.observe(elem);
        });

        // مراقبة الصور (تأثير الستارة)
        document.querySelectorAll('.img-reveal-container').forEach((elem) => {
            observer.observe(elem);
        });

        // محاكاة إرسال النموذج
        function submitForm() {
            const btn = document.querySelector('#bookingForm button[type="submit"] span');
            const originalText = btn.innerText;
            
            btn.innerText = 'جاري الإرسال...';
            
            setTimeout(() => {
                document.getElementById('bookingForm').reset();
                document.getElementById('successMessage').classList.remove('hidden');
                btn.innerText = originalText;
                
                setTimeout(() => {
                    document.getElementById('successMessage').classList.add('hidden');
                }, 4000);
            }, 1500);
        }
    </script>
